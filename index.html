This logic does that automatically. if(introText) { const textContent = introText.innerText; introText.innerHTML = ""; // Clear existing text const words = textContent.split(" "); words.forEach(word => { const span = document.createElement("span"); span.textContent = word + " "; span.style.display = "inline-block"; // Make spans behave for animation introText.appendChild(span); }); gsap.from(introText.children, { opacity: 0.3, y: 20, stagger: 0.05, ease: "power3.out", scrollTrigger: { trigger: introText, start: "top 80%", toggleActions: "play none none none" } }); } });
